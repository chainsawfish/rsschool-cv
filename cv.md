function revers(text){
    let reversStr = ""
    for (let i=text.length-1; i>=0; i-=1){
        reversStr += text[i]
    }
    return reversStr
}



function spinWords(string){
    let newArr = []
    let arrStr = string.split(" ")
    for (let i = 0; i<arrStr.length; i+=1){
        let word = arrStr[i]
        if (word.length > 4) {
            newArr.push(revers(word)) 
            
        }
        else {
            newArr.push(word)
        }
    }

    return newArr.join(" ")
}
